create database TCC;
use TCC;


create table tb_login(
	id_login 	int primary key auto_increment,
    ds_username	varchar(255)	not null,
    ds_senha	varchar(255)	not null,
    ds_perfil	varchar(255)	not null
);

create table tb_cliente(
	id_cliente 					int primary key auto_increment,
    nm_cliente					varchar(255),
    ds_cpf						varchar(255),
    ds_email					varchar(255),
    ds_celular					varchar(30),
    ds_endereco					varchar(255),
    ds_cep						varchar(15),
    ds_numero_endereco			varchar(25),
    id_login				int,
    foreign key (id_login) references tb_login(id_login) on delete cascade
);

create table tb_pedido(
	id_pedido                   int primary key auto_increment,
    nm_pedido                   varchar(255),
    nm_rua                      varchar(255),
    nr_cep                      int,
    id_cliente                  int,
	id_cardapio                 int,
    foreign key (id_cliente) references tb_cliente(id_cliente) on delete cascade,
    foreign key (id_cardapio) references tb_cardapio(id_cardapio) on delete cascade
);
    
create table tb_funcionario(
     id_funcionario               int primary key auto_increment,
     nm_funcionario               varchar(255),
     ds_carteira_trabalho         varchar(50), 
	 ds_email				      varchar(255),
     dt_nascimento			      datetime,
     id_login				      int,
    foreign key (id_login) references tb_login(id_login) on delete cascade
);

create table tb_cardarpio(
      id_cardapio int primary key auto_increment, 
      nm_combo                    varchar (255),
      vl_preco                    decimal (15.2)
);

create table tb_combo(
      id_combo          int primary key auto_increment,
      nm_combo          varchar (255),
      nm_lanche         varchar(255),
      nm_bebida         varchar (255),
      nm_acompanhamento   varchar (255)
);
